<script>
  const roles = ["ETE Student @ RUET", "Data Analyst", "YouTuber", "Problem Solver"];
  let i = 0, j = 0;
  const typed = document.getElementById("typed");

  function type() {
    if (j < roles[i].length) {
      typed.textContent += roles[i][j++];
      setTimeout(type, 80);
    } else {
      setTimeout(erase, 1500);
    }
  }

  function erase() {
    if (j > 0) {
      typed.textContent = roles[i].slice(0, --j);
      setTimeout(erase, 40);
    } else {
      i = (i + 1) % roles.length;
      setTimeout(type, 300);
    }
  }

  type();
</script>
